// RISC-V 32I Assembler Test Program
// Calculates the sum of 1 to 5 and tests instruction formats

_start:
    addi sp, zero, 0x100      # Set Stack Pointer to 256 (tests hex immediate)
    addi a0, zero, 5          # Loop limit = 5
    addi a1, zero, 0          # Current counter (i)
    addi a2, zero, 0          # Running sum

_loop:
    bge a1, a0, _exit         # If i >= 5, branch to _exit (B-Type)

    add a2, a2, a1            # sum = sum + i (R-Type)

    // Test Memory Instructions
    slli t0, a1, 2            # Multiply i by 4 to get word offset (Shift I-Type)
    add t1, sp, t0            # Add offset to base address
    sw a2, 0(t1)              # Store current sum into memory (S-Type)
    lw t2, 0(t1)              # Load it right back to test LW (Memory I-Type)

    // Increment and jump
    addi a1, a1, 1            # i++
    jal zero, _loop           # Jump back to _loop (J-Type)

_exit:
    // Test remaining ALU operations
    sub t3, a2, a1            # Test SUB
    xori t4, t3, -1           # Test negative immediate
    and t5, t4, a2            # Test AND
    or t6, t5, a0             # Test OR
    srli a3, t6, 1            # Test Logical Right Shift

    // Test Jump Register
    jalr zero, 0(ra)          # Halt / Return (I-Type Jump)
