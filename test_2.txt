_return_here  :
    jalr    x0 , -2048( ra)         // Negative offset with weird spacing

_skip_ahead:
    # B-Type: Branches with mixed aliases and x-registers
    beq x1,x2, _start

blt a0, a1, _start
   bge s0, s1, _forward

    lb    x13, -2048 ( x2 )

 lhu x17 , -0x10 (x6 )              # Negative hex literal

    # Memory Stores (S-Type)

     sh x15, -10( x16 )
sw x17,   2047(  x0)

   xori x4, x4, 017                 # octal 15 (if supported, else standard 17)
    ori x5, x5, 2047
 andi x6, x6, -2048

    slli x7, x7, 31

  srai x9, x9, 0b1010               # 10 in binary

 sll x16, x17, x18
    slt x19, x20, x21
    sltu x22, x23, x24
    xor x25, x26, x27
  srl x28, x29, x30
    sra x31, x0, x1
    or x2, x3, x4
  and x5, x6, x7
